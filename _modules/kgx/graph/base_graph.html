[docs]    def edges_iter(self) -> Generator:
        """
        Get an iterable to traverse through all the edges in a graph.

        Returns
        -------
        Generator
            A generator for edges

        """
        for u, v, k, data in self.edges(keys=True, data=True):
            yield (u, v, k, data)
